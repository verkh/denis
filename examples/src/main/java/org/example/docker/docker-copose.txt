
image: - принимает название имаджа, а так же версии
    image: postgres:11.11

restart: - если что-то случилось с контейнером, он аварийно завершился, то с ключем always, контейнер будет перезапускаться

environment: - перменные окружения

ports: - маппинг портов, чтобы они были доступны извне контейнера
        <порт на машине>:<порт внутри контейнера>

v


docker-compose up - запустить котейнеры из файла в текущей дирректории docker-compose.yml

docker-compose -f <filepath> up - запустить контейнеры из указанного файла

docker-compose down --remove-orphans - удалить все контейнеры из файла
